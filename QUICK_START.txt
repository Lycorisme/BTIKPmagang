   - Import file SQL (jalankan query ini):

   -------- COPY MULAI DARI SINI --------
   CREATE DATABASE IF NOT EXISTS magang_db;
   USE magang_db;

   CREATE TABLE users (
       id INT AUTO_INCREMENT PRIMARY KEY,
       nama VARCHAR(100) NOT NULL,
       email VARCHAR(100) NOT NULL UNIQUE,
       password VARCHAR(255) NOT NULL,
       role ENUM('mahasiswa', 'mentor', 'admin') NOT NULL,
       created_at DATETIME DEFAULT CURRENT_TIMESTAMP
   );

   CREATE TABLE mentors (
       user_id INT PRIMARY KEY,
       keahlian VARCHAR(255) NOT NULL,
       bio TEXT,
       status_open BOOLEAN DEFAULT 1,
       foto VARCHAR(255),
       FOREIGN KEY (user_id) REFERENCES users(id)
   );

   CREATE TABLE lowongan (
       id INT AUTO_INCREMENT PRIMARY KEY,
       judul VARCHAR(255) NOT NULL,
       deskripsi TEXT,
       mentor_id INT NOT NULL,
       tgl_mulai DATE,
       tgl_selesai DATE,
       status ENUM('open', 'closed') DEFAULT 'open',
       FOREIGN KEY (mentor_id) REFERENCES mentors(user_id)
   );

   CREATE TABLE lamaran (
       id INT AUTO_INCREMENT PRIMARY KEY,
       user_id INT NOT NULL,
       lowongan_id INT NOT NULL,
       mentor_id INT NOT NULL,
       tgl_melamar DATE NOT NULL,
       status ENUM('proses', 'diterima', 'ditolak') DEFAULT 'proses',
       file_cv VARCHAR(255),
       FOREIGN KEY (user_id) REFERENCES users(id),
       FOREIGN KEY (lowongan_id) REFERENCES lowongan(id),
       FOREIGN KEY (mentor_id) REFERENCES mentors(user_id)
   );

   CREATE TABLE jurnal (
       id INT AUTO_INCREMENT PRIMARY KEY,
       user_id INT NOT NULL,
       mentor_id INT NOT NULL,
       tanggal DATE NOT NULL,
       aktivitas TEXT,
       file_penunjang VARCHAR(255),
       feedback TEXT,
       nilai FLOAT,
       FOREIGN KEY (user_id) REFERENCES users(id),
       FOREIGN KEY (mentor_id) REFERENCES mentors(user_id)
   );
   -------- COPY SAMPAI SINI --------
